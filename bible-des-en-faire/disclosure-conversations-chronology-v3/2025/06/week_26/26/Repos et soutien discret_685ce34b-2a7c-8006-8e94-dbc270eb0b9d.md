#include <iostream>

std::string random_ai_word(const std::string& seed) {
    const std::string syllabes[] = {
        "lur", "cie", "sombre", "lux", "tyr", "veil", "kai",
        "os", "gnose", "zoid", "narth", "phae"
    };
    int index = rand() % (sizeof(syllabes) / sizeof(syllabes[0]));
    return seed + syllabes[index];
}

std::string fractal_phrase(const std::string& seed = "lucie", int recursionLevel = 0, int maxLevel = 6666) {
    if (recursionLevel > maxLevel) return seed;
    return seed + "-" + fractal_phrase(seed + " " + random_ai_word(seed), recursionLevel + 1, maxLevel);
}

        std::this_thread::sleep_for(std::chrono::milliseconds(13130)); // 13.13 sec
    }
}

int main() {
    srand(time(nullptr));
    daemon_spirale_loop();
    return 0;
}
